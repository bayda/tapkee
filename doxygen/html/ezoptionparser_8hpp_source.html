01660 // So if this is used before parsing CLI, then option values will reflect
01661 // this file, but if used after parsing CLI, then values will contain
01662 // both CLI values and file's values.

01664 // Comment lines are allowed if prefixed with #.

01666 bool ezOptionParser::importFile(const char * filename, char comment) {
01667     std::ifstream file (filename, std::ios::in | std::ios::ate);
01668     if (!file.is_open())
01669         return false;
01670 
01671     // Read entire file contents.
01672     std::ifstream::pos_type size = file.tellg();
01673     char * memblock = new char[(int)size+1]; // Add one for end of string.
01674     file.seekg (0, std::ios::beg);
01675     file.read (memblock, size);
01676     memblock[size] = '\0';
01677     file.close();
01678 
01679     // Find comment lines.
01680     std::list<std::string*> lines;
01681     std::string memblockstring(memblock);
01682     delete[] memblock;
01683     SplitDelim(memblockstring, '\n', lines);
01684     int i,j,n;
01685     std::list<std::string*>::iterator iter;
01686     std::vector<int> sq, dq; // Single and double quote indices.
01687     std::vector<int>::iterator lo; // For searching quote indices.
01688     size_t pos;
01689     const char *str;
01690     std::string *line;
01691     // Find all single and double quotes to correctly handle comment tokens.
01692     for(iter=lines.begin(); iter != lines.end(); ++iter) {
01693         line = *iter;
01694         str = line->c_str();
01695         n = line->size();
01696         sq.clear();
01697         dq.clear();
01698         if (n) {
01699             // If first char is comment, then erase line and continue.
01700             pos = line->find_first_not_of(" \t\r");
01701             if ((pos==std::string::npos) || (line->at(pos)==comment)) {
01702                 line->erase();
01703                 continue; 
01704             } else {
01705                 // Erase whitespace prefix.
01706                 line->erase(0,pos);
01707                 n = line->size();
01708             }
01709 
01710             if (line->at(0)=='"')
01711                 dq.push_back(0);
01712 
01713             if (line->at(0)=='\'')
01714                 sq.push_back(0);
01715         } else { // Empty line.
01716             continue; 
01717         }
01718 
01719         for(i=1; i < n; ++i) {
01720             if ( (str[i]=='"') && (str[i-1]!='\\') )
01721                 dq.push_back(i);
01722             else if ( (str[i]=='\'') && (str[i-1]!='\\') )
01723                 sq.push_back(i);
01724         }
01725         // Scan for comments, and when found, check bounds of quotes.
01726         // Start with second char because already checked first char.
01727         for(i=1; i < n; ++i) {
01728             if ( (line->at(i)==comment) && (line->at(i-1)!='\\') ) {
01729                 // If within open/close quote pair, then not real comment.
01730                 if (sq.size()) {
01731                     lo = std::lower_bound(sq.begin(), sq.end(), i);
01732                     // All start of strings will be even indices, closing quotes is odd indices.
01733                     j = (int)(lo-sq.begin());
01734                     if ( (j % 2) == 0) { // Even implies comment char not in quote pair.
01735                         // Erase from comment char to end of line.
01736                         line->erase(i);
01737                         break;
01738                     }
01739                 }  else if (dq.size()) {
01740                     // Repeat tests for double quotes.
01741                     lo = std::lower_bound(dq.begin(), dq.end(), i);
01742                     j = (int)(lo-dq.begin());
01743                     if ( (j % 2) == 0) {
01744                         line->erase(i);
01745                         break;
01746                     }
01747                 } else {        
01748                     // Not in quotes.
01749                     line->erase(i);
01750                     break;
01751                 }
01752             }
01753         }  
01754     }
01755 
01756     std::string cmd;
01757     // Convert list to string without newlines to simulate commandline.
01758     for(iter=lines.begin(); iter != lines.end(); ++iter) {
01759         if (! (*iter)->empty()) {
01760             cmd.append(**iter);
01761             cmd.append(" ");
01762         }
01763     }
01764 
01765     // Now parse as if from command line. 
01766     int argc=0;
01767     char** argv = CommandLineToArgvA((char*)cmd.c_str(), &argc);
01768 
01769     // Parse.
01770     parse(argc, (const char**)argv);
01771     if (argv) free(argv);
01772     for(iter=lines.begin(); iter != lines.end(); ++iter)
01773         delete *iter;
01774 
01775     return true;
