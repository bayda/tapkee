00161     bool insert(int new_index)
00162     {
00163         // Ignore objects which do not belong in this quad tree
00164         double* point = data + new_index * QT_NO_DIMS;

00166             return false;
00167         
00168         // Online update of cumulative size and center-of-mass
00169         cum_size++;

00171         double mult2 = 1.0 / (double) cum_size;
00172         for(int d = 0; d < QT_NO_DIMS; d++) center_of_mass[d] *= mult1;
00173         for(int d = 0; d < QT_NO_DIMS; d++) center_of_mass[d] += mult2 * point[d];

00297     // Compute non-edge forces using Barnes-Hut algorithm
00298     void computeNonEdgeForces(int point_index, double theta, double neg_f[], double* sum_Q)
00299     {
00300         
00301         // Make sure that we spend no time on empty nodes or self-interactions
00302         if(cum_size == 0 || (is_leaf && size == 1 && index[0] == point_index)) return;
00303         
00304         // Compute distance between point and center-of-mass
00305         double D = .0;
00306         int ind = point_index * QT_NO_DIMS;
00307         for(int d = 0; d < QT_NO_DIMS; d++) buff[d]  = data[ind + d];
00308         for(int d = 0; d < QT_NO_DIMS; d++) buff[d] -= center_of_mass[d];
00309         for(int d = 0; d < QT_NO_DIMS; d++) D += buff[d] * buff[d];
00310         
00311         // Check whether we can use this node as a "summary"
00312         if(is_leaf || max(boundary.hh, boundary.hw) / sqrt(D) < theta) {
00313         
00314             // Compute and add t-SNE force between point and current node
00315             double Q = 1.0 / (1.0 + D);
00316             *sum_Q += cum_size * Q;
00317             double mult = cum_size * Q * Q;
00318             for(int d = 0; d < QT_NO_DIMS; d++) neg_f[d] += mult * buff[d];
00319         }
00320         else {
00321 
00322             // Recursively apply Barnes-Hut to children
00323             northWest->computeNonEdgeForces(point_index, theta, neg_f, sum_Q);
00324             northEast->computeNonEdgeForces(point_index, theta, neg_f, sum_Q);
00325             southWest->computeNonEdgeForces(point_index, theta, neg_f, sum_Q);
00326             southEast->computeNonEdgeForces(point_index, theta, neg_f, sum_Q);
00327         }
00328     }
