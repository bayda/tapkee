00257             select_landmarks_random(begin,end,ratio);
00258         DenseMatrix distance_matrix = 
00259             compute_shortest_distances_matrix(begin,end,landmarks,neighbors,distance_callback);
00260         distance_matrix = distance_matrix.array().square();
00261         
00262         DenseVector col_means = distance_matrix.colwise().mean();
00263         DenseVector row_means = distance_matrix.rowwise().mean();
00264         ScalarType grand_mean = distance_matrix.mean();
00265         distance_matrix.array() += grand_mean;
00266         distance_matrix.colwise() -= row_means;
00267         distance_matrix.rowwise() -= col_means.transpose();
00268         distance_matrix.array() *= -0.5;
00269 
00270         EmbeddingResult landmarks_embedding = 
00271             eigen_embedding<DenseSymmetricMatrix,DenseMatrixOperation>(eigen_method,
00272                     distance_matrix*distance_matrix.transpose(),target_dimension,SKIP_NO_EIGENVALUES);
00273 
00274         DenseMatrix embedding = distance_matrix.transpose()*landmarks_embedding.first;
00275 
00276         for (IndexType i=0; i<target_dimension; i++)
00277             embedding.col(i).array() /= sqrt(sqrt(landmarks_embedding.second(i)));
00278         return ReturnResult(embedding,tapkee::ProjectingFunction());
