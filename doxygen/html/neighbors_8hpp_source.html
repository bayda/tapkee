00048 template <class RandomAccessIterator, class PairwiseCallback>
00049 Neighbors find_neighbors_covertree_impl(RandomAccessIterator begin, RandomAccessIterator end, 
00050                          PairwiseCallback callback, IndexType k)
00051 {
00052     timed_context context("Covertree-based neighbors search");
00053 
00054     typedef CoverTreePoint<RandomAccessIterator> TreePoint;
00055     v_array<TreePoint> points;
00056     for (RandomAccessIterator iter=begin; iter!=end; ++iter)
00057         push(points, TreePoint(iter, callback(*iter,*iter)));
00058 
00059     node<TreePoint> ct = batch_create(callback, points);
00060 
00061     v_array< v_array<TreePoint> > res;
00062     ++k; // because one of the neighbors will be the actual query point
00063     k_nearest_neighbor(callback,ct,ct,res,k);
00064 
00065     Neighbors neighbors;
00066     neighbors.resize(end-begin);
00067     assert(end-begin==res.index);
00068     for (int i=0; i<res.index; ++i)
00069     {
00070         LocalNeighbors local_neighbors;
00071         local_neighbors.reserve(k);
00072         
00073         for (IndexType j=1; j<=k; ++j) // j=0 is the query point
00074         {
00075             // The actual query point is found as a neighbor, just ignore it
00076             if (res[i][j].iter_-begin==res[i][0].iter_-begin)
00077                 continue;
00078             local_neighbors.push_back(res[i][j].iter_-begin);
00079         }
00080         neighbors[res[i][0].iter_-begin] = local_neighbors;
00081         free(res[i].elements);
00082     };
00083     free(res.elements);
00084     free_children(ct);
00085     free(points.elements);
00086     return neighbors;
00087 }

00090 Neighbors find_neighbors_bruteforce_impl(const RandomAccessIterator& begin, const RandomAccessIterator& end, 
00091                                          PairwiseCallback callback, IndexType k)
00092 {
00093     timed_context context("Distance sorting based neighbors search");
00094     typedef std::pair<RandomAccessIterator, ScalarType> DistanceRecord;
00095     typedef std::vector<DistanceRecord> Distances;
00096 
00097     Neighbors neighbors;
00098     neighbors.reserve(end-begin);
00099     for (RandomAccessIterator iter=begin; iter!=end; ++iter)
00100     {
00101         Distances distances;
00102         if (BasicCallbackTraits<PairwiseCallback>::is_kernel())
00103         {
00104             for (RandomAccessIterator around_iter=begin; around_iter!=end; ++around_iter)
00105                 distances.push_back(make_pair(around_iter, callback(*around_iter,*around_iter) + callback(*iter,*iter) - 2*callback(*iter,*around_iter)));
00106         }
00107         else
00108         {
00109             for (RandomAccessIterator around_iter=begin; around_iter!=end; ++around_iter)
00110                 distances.push_back(make_pair(around_iter, callback(*iter,*around_iter)));
00111         }
00112 
00113         nth_element(distances.begin(),distances.begin()+k+1,distances.end(),
00114                     distances_comparator<DistanceRecord>());
00115 
00116         LocalNeighbors local_neighbors;
00117         local_neighbors.reserve(k);
00118         for (typename Distances::const_iterator neighbors_iter=distances.begin()+1; 
00119                 neighbors_iter!=distances.begin()+k+1; ++neighbors_iter)
00120             local_neighbors.push_back(neighbors_iter->first - begin);
00121         neighbors.push_back(local_neighbors);
00122     }
00123     return neighbors;
