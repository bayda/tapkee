00140                 ScalarType v1 = (ScalarType)(rand()+1.f)/((float)RAND_MAX+2.f);
00141                 ScalarType v2 = (ScalarType)(rand()+1.f)/((float)RAND_MAX+2.f);
00142                 ScalarType len = sqrt(-2.f*log(v1));
00143                 O(i,j) = len*cos(2.f*M_PI*v2);
00144             }
00145         }
00146         MatrixOperationType operation(wm);
00147 
00148         DenseMatrix Y = operation(O);
00149         for (IndexType i=0; i<Y.cols(); i++)
00150         {
00151             for (IndexType j=0; j<i; j++)
00152             {
00153                 ScalarType r = Y.col(i).dot(Y.col(j));
00154                 Y.col(i) -= r*Y.col(j);
00155             }
00156             ScalarType norm = Y.col(i).norm();

00158             {
00159                 for (int k = i; k<Y.cols(); k++)
00160                     Y.col(k).setZero();

00170         {
00171             if (MatrixOperationType::largest)
00172             {
00173                 assert(skip==0);
