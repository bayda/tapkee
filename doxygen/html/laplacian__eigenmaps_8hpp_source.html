00038 Laplacian compute_laplacian(RandomAccessIterator begin, 
00039             RandomAccessIterator end,const Neighbors& neighbors, 
00040             DistanceCallback callback, ScalarType width)
00041 {
00042     SparseTriplets sparse_triplets;
00043 
00044     timed_context context("Laplacian computation");
00045     const IndexType k = neighbors[0].size();
00046     sparse_triplets.reserve((k+1)*(end-begin));
00047 
00048     DenseVector D = DenseVector::Zero(end-begin);
00049     for (RandomAccessIterator iter=begin; iter!=end; ++iter)
00050     {
00051         const LocalNeighbors& current_neighbors = neighbors[iter-begin];
00052 
00053         for (IndexType i=0; i<k; ++i)
00054         {
00055             ScalarType distance = callback(*iter,begin[current_neighbors[i]]);
00056             ScalarType heat = exp(-distance*distance/width);
00057             D(iter-begin) += heat;
00058             D(current_neighbors[i]) += heat;
00059             sparse_triplets.push_back(SparseTriplet(current_neighbors[i],(iter-begin),-heat));
00060             sparse_triplets.push_back(SparseTriplet((iter-begin),current_neighbors[i],-heat));
00061         }
00062     }
00063     for (IndexType i=0; i<(end-begin); ++i)
00064         sparse_triplets.push_back(SparseTriplet(i,i,D(i)));
00065 
00066 #ifdef EIGEN_YES_I_KNOW_SPARSE_MODULE_IS_NOT_STABLE_YET
00067     Eigen::DynamicSparseMatrix<ScalarType> dynamic_weight_matrix(end-begin,end-begin);
00068     dynamic_weight_matrix.reserve(sparse_triplets.size());
00069     for (SparseTriplets::const_iterator it=sparse_triplets.begin(); it!=sparse_triplets.end(); ++it)
00070         dynamic_weight_matrix.coeffRef(it->col(),it->row()) += it->value();
00071     SparseWeightMatrix weight_matrix(dynamic_weight_matrix);
00072 #else
00073     SparseWeightMatrix weight_matrix(end-begin,end-begin);
00074     weight_matrix.setFromTriplets(sparse_triplets.begin(),sparse_triplets.end());
00075 #endif
